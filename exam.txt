#!/bin/bash
# Script to prompt for a file path and display its properties on a single line.

read -p "Enter a File (including path): " filepath

# Use ls -ld to get the long listing of the file/directory itself.
# Store the output in a variable.
output=$(ls -ld "$filepath")

# Print the required prefix and the output.
# This results in the properties being displayed on a single line.
echo "Properties: $output"
